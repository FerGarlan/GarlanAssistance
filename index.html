    <script>

        const MASTER_BENEFITS = ["Asistencia médica por accidente y/o enfermedad no preexistente", "Asistencia médica por enfermedad preexistente", "Acompañamiento de menores y/o mayores", "Emergencia dental", "Medicamentos", "Medicamentos en caso de hospitalizacion", "Cancelación y/o interrupción de viaje", "Gastos médicos por complicaciónes de embarazo", "Gastos por vuelo demorado o Cancelado (más de 6 hs)", "Pérdida de vuelo", "Prótesis y órtesis", "Regreso anticipado por siniestro grave en domicilio", "Regreso de acompañanate de Titular Repatriado", "Traslado de familiar por hospitalización", "Gastos de hotel familiar acompañante", "Viaje de regreso por fallecimiento de familiar en 1°", "Traslado sanitario / Repatriación Sanitaria", "Repatriación de restos mortales", "Localización de equipaje", "Seguro de pérdida de equipaje (Complementario)", "Gastos por demora de entrega de equipaje (+ de 8hs)", "Transferencia de fondos", "Asistencia Legal por accidente de tránsito"];

       

        const initialPlans = [

            { id: 1, name: "Garlan Lite", price: 6, duration: "Por Día", active: true, benefits: MASTER_BENEFITS.map(b => ({ label: b, value: "USD 10.000" })) },

            { id: 4, name: "Garlan Elite", price: 18, duration: "Por Día", active: true, benefits: MASTER_BENEFITS.map(b => ({ label: b, value: "USD 100.000" })) }

        ];

 

        let plans = JSON.parse(localStorage.getItem('garlan_data')) || initialPlans;

        let isAdmin = false;

 

        function showPage(pageId) {

            document.querySelectorAll('.page-view').forEach(p => p.classList.remove('active'));

            document.getElementById(pageId).classList.add('active');

            if(pageId === 'plans') renderPublicPlans();

            if(pageId === 'admin') renderAdminTable();

            window.scrollTo(0,0);

        }

 

        function handleLogin(e) {

            e.preventDefault();

            if(document.getElementById('username').value === 'admin') {

                isAdmin = true;

                document.getElementById('adminBtn').classList.add('hidden');

                document.getElementById('logoutBtn').classList.remove('hidden');

                showPage('admin');

            }

        }

 

        function logout() {

            isAdmin = false;

            document.getElementById('adminBtn').classList.remove('hidden');

            document.getElementById('logoutBtn').classList.add('hidden');

            showPage('home');

        }

 

        function renderPublicPlans() {

            const grid = document.getElementById('plansGrid');

            grid.innerHTML = plans.filter(p => p.active).map(p => `

                <div class="plan-card bg-white rounded-2xl overflow-hidden flex flex-col h-full">

                    <div class="p-6 bg-slate-900 text-white">

                        <span class="text-[10px] font-bold uppercase tracking-widest px-2 py-1 bg-blue-600 rounded mb-4 inline-block">${p.duration}</span>

                        <h3 class="text-xl font-extrabold">${p.name}</h3>

                        <div class="mt-4"><span class="text-3xl font-black">$${p.price}</span> <span class="text-slate-400 text-xs font-bold">USD</span></div>

                    </div>

                    <div class="p-6 flex-1 flex flex-col">

                        <div class="space-y-2 mb-6 max-h-64 overflow-y-auto pr-2 modal-scroll text-[11px]">

                            ${p.benefits.map(b => `<div class="flex justify-between pb-1 border-b border-slate-50"><span>${b.label}</span><span class="font-bold">${b.value}</span></div>`).join('')}

                        </div>

                        <button onclick="showPage('home')" class="w-full btn-primary text-white py-3 rounded-lg text-sm font-bold mt-auto">Comprar</button>

                    </div>

                </div>`).join('');

        }

 

        function renderAdminTable() {

            const table = document.getElementById('adminTable');

            table.innerHTML = plans.map(p => `

                <tr class="${!p.active ? 'opacity-50' : ''}">

                    <td class="p-4 font-bold">${p.name}</td>

                    <td class="p-4 text-xs">${p.duration}</td>

                    <td class="p-4 font-bold text-blue-700">$${p.price}</td>

                    <td class="p-4 text-right flex justify-end gap-3">

                        <button onclick="togglePlanStatus(${p.id})" class="font-bold text-xs uppercase">${p.active ? 'Inactivar' : 'Activar'}</button>

                        <button onclick="editPlan(${p.id})" class="text-blue-600 font-bold text-xs uppercase">Editar</button>

                    </td>

                </tr>`).join('');

        }

 

        function togglePlanStatus(id) {

            const idx = plans.findIndex(p => p.id === id);

            if (idx !== -1) {

                plans[idx].active = !plans[idx].active;

                localStorage.setItem('garlan_data', JSON.stringify(plans));

                renderAdminTable();

            }

        }

 

        function generateBenefitInputs(planBenefits = []) {

            const container = document.getElementById('benefitList');

            container.innerHTML = "";

            MASTER_BENEFITS.forEach(bName => {

                const exist = planBenefits.find(b => b.label === bName);

                const val = exist ? exist.value : "";

                const div = document.createElement('div');

                div.className = "flex flex-col bg-slate-50 p-2 rounded-lg border";

                div.innerHTML = `<label class="text-[9px] font-bold text-slate-400 uppercase mb-1">${bName}</label><input type="text" data-label="${bName}" value="${val}" class="benefit-input w-full bg-white p-1.5 rounded text-[11px] font-bold text-blue-700 outline-none border border-slate-200">`;

                container.appendChild(div);

            });

        }

 

        function editPlan(id) {

            const p = plans.find(x => x.id === id);

            if (!p) return;

            document.getElementById('editId').value = p.id;

            document.getElementById('editName').value = p.name;

            document.getElementById('editPrice').value = p.price;

            document.getElementById('editDuration').value = p.duration;

            generateBenefitInputs(p.benefits);

            document.getElementById('modal').classList.remove('hidden');

        }

 

        function savePlan(e) {

            e.preventDefault();

            const id = document.getElementById('editId').value;

            const bInputs = document.querySelectorAll('.benefit-input');

            const benefits = Array.from(bInputs).map(inp => ({ label: inp.getAttribute('data-label'), value: inp.value.trim() || "Incluido" }));

            const data = { id: id ? parseInt(id) : Date.now(), name: document.getElementById('editName').value, price: parseFloat(document.getElementById('editPrice').value), duration: document.getElementById('editDuration').value, active: true, benefits: benefits };

            if(id) { const idx = plans.findIndex(x => x.id === parseInt(id)); plans[idx] = data; } else { plans.push(data); }

            localStorage.setItem('garlan_data', JSON.stringify(plans));

            closePlanModal();

            renderAdminTable();

        }

 

        function closePlanModal() { document.getElementById('modal').classList.add('hidden'); }

        function openPlanModal() { document.getElementById('editId').value = ""; document.getElementById('planForm').reset(); generateBenefitInputs([]); document.getElementById('modal').classList.remove('hidden'); }

       

        window.onload = () => showPage('home');

    </script>
